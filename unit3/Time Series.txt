library(forecast)

ts <- ts(R_data$Demand, frequency = 12, start = 2012)
plot(ts)

tsdA <- decompose(ts, "additive")
plot(tsdA)

tsdM <- decompose(ts, "multiplicative")
plot(tsdM)

#Naive

NaiveModel <- naive(ts, h = 1)
summary(NaiveModel)

#Cumulative

CumulativeModel <- meanf(ts, h = 1)
summary(CumulativeModel)


#Moving average

MA2Model <- ma(ts, order = 2)

MA2Model 

MA4Model <- ma(ts, order = 4)

MA4Model 


#Holt model
Hts <- ts(holt_sample$Xt, frequency = 12, start = 2019)

HoltModel <- holt(Hts, h=6, alpha=0.2, beta=0.1)

summary(HoltModel)

plot(HoltModel)


#HoltWinters model

HWModelA <- hw(Hts, h=12, alpha=0.2, beta=0.1, gamma=0.25, seasonal = "additive")

summary(HWModelA)

plot(HWModelA)

HWModelM <- hw(Hts, h=12, alpha=0.2, beta=0.1, gamma=0.25, seasonal = "multiplicative")

summary(HWModelM)

plot(HWModelM)
